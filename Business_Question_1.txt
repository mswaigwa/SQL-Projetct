WITH sales_person AS (SELECT first_name, last_name, employee_ID 
                      FROM HR.Employees 
                      WHERE Department_ID =80),
sales AS (SELECT SUM (Order_Total), Sales_Rep_ID 
          FROM OE.Orders
          GROUP BY Sales_Rep_ID, EXTRACT(YEAR FROM order_date)
          HAVING SUM(Order_Total)>250000 
          AND EXTRACT(YEAR FROM order_date) = 2007
          AND Sales_Rep_ID IS NOT NULL)
SELECT p.*,s.* 
FROM sales_person p
JOIN sales s
ON sales_Rep_ID = employee_ID
